//model relationShip: (study db relationships)

one to many(user can have many posts)
example:
model User{
    id Int @id @default(autoincrement())
    name string 
    email string
    preferences Json?
    post Post[] // [] is an modifier it means there can be many posts, ? means optional field
}

one to many ()

model Post{
    rating float
    author User @relationShip(fields:[userId],references:[id])//fields in out id references which id in User table and references is our foreign key
    userId : Int
    authorId String

}
